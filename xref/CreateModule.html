183     private void write_byte(int location, int value) {
184         module[location] = (byte) (value & 0xff);
185     }

187     private void write_word(int location, int value) {
188         byte highByte = (byte) ((value >> 8) & 0xff);
189         module[location] = highByte;
190         byte lowByte = (byte) (value & 0xff);
191         module[location + 1] = lowByte;
192     }

194     private void writeString(int location, String name) {
195         byte[] nameBytes = name.getBytes();
196 
197         for (int i = 0; i < nameBytes.length; i++) {
198             module[location + i] = nameBytes[i];
199         }
200         module[location + nameBytes.length - 1] |= 0x80;
201     }

203     private int calculateModuleSize(int fileSize) {
204         int headerSize = 12;
205         if (moduleType == 1) {
206             headerSize += 4;
207         }
208         headerSize += moduleName.length();
209

213     private void writeParity() {
214         byte parity = 0;
215         for (int i = 0; i < 8; i++) {
216             parity = (byte) (parity ^ module[i]);
217         }
218         module[8] = parity;
219     }

221     private void writeCRC(int location) {
222         long crc = OS9.CRC24_SEED;
223         long result = OS9.compute_crc(crc, module, location) ^ OS9.CRC24_SEED;
224         module[location + 0] = (byte) ((result >> 16) & 0xFF);
225         module[location + 1] = (byte) ((result >> 8) & 0xFF);
226         module[location + 2] = (byte) (result & 0xFF);
227     }
