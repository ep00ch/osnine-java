65              fd = new PDUnix(realPath, mode, create);
66              fd.setDriver(this);
67          } catch (Exception e) {
68              errorcode = ErrCodes.E_BPNam;

140     static FileAttribute<Set<PosixFilePermission>> o2u_attr(int omode) {
141         TreeSet<PosixFilePermission> umode = new TreeSet<PosixFilePermission>();
142         if ((omode & 1) == 1)  umode.add(PosixFilePermission.OWNER_READ);
143         if ((omode & 2) == 2)  umode.add(PosixFilePermission.OWNER_WRITE);
144         if ((omode & 4) == 4)  umode.add(PosixFilePermission.OWNER_EXECUTE);
145         if ((omode & 8) == 8) {
146             umode.add(PosixFilePermission.GROUP_READ);
147             umode.add(PosixFilePermission.OTHERS_READ);
148         }
149         if ((omode & 16) == 16) {
150             umode.add(PosixFilePermission.GROUP_WRITE);
151             umode.add(PosixFilePermission.OTHERS_WRITE);
152         }
153         if ((omode & 32) == 32) {
154             umode.add(PosixFilePermission.GROUP_EXECUTE);
155             umode.add(PosixFilePermission.OTHERS_EXECUTE);
156         }
157         return PosixFilePermissions.asFileAttribute(umode);
158     }

160     /**
161      * Find the UNIX file from the OS9 path. The OS9 path
162      * is case-insensive, so we have to check every directory.
163      *
164      * @param path - OS9 path.
165      * @param mustexist - ?
166      */
167     static Path findpath(Path path, boolean mustexist) {
168         String startFrom;
169         int alreadyMatched = 0;
170 
171         path = path.normalize();
172         if (path.startsWith("/")) {
173             startFrom = "/";
174         } else {
175             startFrom = ".";
176         }
177         File newUnixFile = new File(startFrom);
178         int segmentCnt = path.getNameCount();
179 
180         boolean found = false;
181         while (alreadyMatched < segmentCnt) {
182             found = false;
183             for (File dirEntry : newUnixFile.listFiles()) {
184                 String nextOS9Seg = path.getName(alreadyMatched).toString();
185                 String nextUNXSeg = dirEntry.getName();
186                 if (nextUNXSeg.equalsIgnoreCase(nextOS9Seg)) {
187                     alreadyMatched++;
188                     newUnixFile = new File(newUnixFile, nextUNXSeg);
189                     found = true;
190                     break;
191                 }
192             }
193             if (!found) {
194                 break;
195             }
196         }
197         LOGGER.debug("Found path: {}", newUnixFile);
198         // Some intermediate segment is missing.
199         if (alreadyMatched < segmentCnt - 1) {
200             return null;
201         }
202         if (mustexist && !found) {
203             return null;
204         }
205         if (!mustexist && !found) {
206             newUnixFile = new File(newUnixFile, path.toFile().getName());
207         }
208         return newUnixFile.toPath();
209     }
