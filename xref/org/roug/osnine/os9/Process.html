35       * Constructor.

37      public Process(OS9 kernel, Process parentProcessObj) {
38          this.kernel = kernel;
39  
40          for (int inx = 0; inx < NUM_PATHS; inx++) {
41              setPathDesc(inx, null);
42          }
43  
44          int tmpx = kernel.x.intValue();
45          int tmpy = kernel.y.intValue();
46          int tmpd = kernel.d.intValue();
47          kernel.x.set(kernel.read_word(DPConst.D_PrcDBT));  // 73- Process descriptor block address
48          kernel.f_all64();                // Allocate the process descriptor
49          kernel.write_word(DPConst.D_PrcDBT, kernel.x.intValue());
50          procAddr = kernel.y.intValue();
51          kernel.x.set(tmpx);
52          kernel.y.set(tmpy);
53          kernel.d.set(tmpd);
54          setProcessId(kernel.getNextPID());
55          if (parentProcessObj == null) {
56              setParentProcessId(0); // No parent process.
57          } else {
58              setParentProcessId(parentProcessObj.getProcessId());
59          }
60      }
