68          int i, r;

71              for (i = 0; i < size; i++) {
72                  r = infp.read(c);
73                  if (r == -1) {
74                      return i;
75                  }
76                  if (c[0] == NEW_LINE) // Do conversion
77                      c[0] = CARRIAGE_RETURN;
78                  buf[i] = c[0];
79                  if (c[0] == CARRIAGE_RETURN) {
80                      i++;
81                      break;
82                  }
83              }
84          } catch (IOException e) {

102      * Only ttys files here.
103      */
104     @Override
105     public int writeln(byte[] buf, int size) {
106         int inx = 0;
107      
108         for (inx = 0; inx < size; inx++) {
109             if (buf[inx] == CARRIAGE_RETURN) {
110                 fp.write(convertToUNIX() ? NEW_LINE : CARRIAGE_RETURN);
111                 inx++;
112                 break;
113             } else {
114                 fp.write(buf[inx]);
115             }
116         }
117         fp.flush();
118         return inx;
119     }

132         int inx;
133 
134         switch (cpu.b.intValue()) {

136             for (inx = 0; inx < 32; inx++)
137                 cpu.write(inx + cpu.x.intValue(), 0);
138             cpu.write(cpu.x.intValue(), 0x0); /* RBF */
139             cpu.write(cpu.x.intValue() + 0x08, 24); /* Lines per page */
140             cpu.write(cpu.x.intValue() + 0x09, 8);  /* BS char */
141             cpu.write(cpu.x.intValue() + 0x0a, 0x7f); /* DEL char */
142             cpu.write(cpu.x.intValue() + 0x0b, 13); /* EOR char */
143             cpu.write(cpu.x.intValue() + 0x0c, 4); /* EOF char ctrl-d */
144             break;
145         case 38: // SS.ScSiz
146             cpu.x.set(80);
147             break;
148         default:
149             cpu.sys_error(ErrCodes.E_UnkSvc);
150         }
