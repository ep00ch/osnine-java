12      public final static int DD_TKS = 0x03;
13      public final static int DD_MAP = 0x04;
14      public final static int DD_BIT = 0x06;
15      public final static int DD_DIR = 0x08;

18      public final static int DD_DSK = 0x0E;
19      public final static int DD_FMT = 0x10;
20      public final static int DD_SPT = 0x11;
21      public final static int DD_RES = 0x13;
22      public final static int DD_BT  = 0x15;
23      public final static int DD_BSZ = 0x18;
24      public final static int DD_DAT = 0x1A;
25      public final static int DD_NAM = 0x1F;
26      public final static int DD_OPT = 0x3F;
27  
28      /** Number of reading heads. Two for floppy disks */
29      private int numHeads = 2;
30  
31      /** Total number of sectors on media. */
32      private int totalSectors;
33  
34      /** Sectors per track. */
35      private int sectorsPerTrack = 0;
36

38      private int clusterSize = 1;

47          }

49              throw new RuntimeException("Too few sectors to allocate");
50          }

52      }

60      }

63          rootDirLSN = s;
64      }

66      void setDiskName(String name) {
67          if (name.length() > 32) {
68              throw new RuntimeException("Disk name can be max 32 characters");
69          }
70          char[] arr = name.toCharArray();
71          for (int i = 0; i < arr.length; i++) {
72              if (arr[i] < ' ' || arr[i] > 127) {
73                  throw new RuntimeException("Only ASCII is allowed");
74              }
75          }
76          diskName = name;
77      }
78  
79      public byte[] getSector() {
80          SectorSupport sector = new SectorSupport();
81          int bytesInAllocMap = totalSectors / 8 / clusterSize;
82  
83          sector.writeTriple(DD_TOT, totalSectors);
84          sector.writeByte(DD_TKS, sectorsPerTrack);
85          sector.writeWord(DD_MAP, bytesInAllocMap );
86  
87          sector.writeWord(DD_BIT, 1 << (clusterSize - 1));
88          // Write FD sector of root directory.
89          sector.writeTriple(DD_DIR, rootDirLSN);
90          sector.writeWord(DD_OWN, 0); // Owners user number.
91          sector.writeByte(DD_ATT, 0xFF);  // Disk attributes
92          int randomNum = ThreadLocalRandom.current().nextInt(0, 0x10000);
93          sector.writeWord(DD_DSK, randomNum);
94          sector.writeByte(DD_FMT, 2 + (numHeads - 1));
95          sector.writeWord(DD_SPT, sectorsPerTrack);
96          sector.writeWord(DD_RES, 0); // Reserved
97          sector.writeTriple(DD_BT, 0);  // Boot sector
98          sector.writeWord(DD_BSZ, 0); // Size of boot sector
99          sector.writeDate(DD_DAT);
100         sector.writeString(DD_NAM, diskName);
101 
102         return sector.getSector();
