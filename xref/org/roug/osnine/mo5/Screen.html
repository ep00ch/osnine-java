257     /**
258      * Change the pixel buffer at the given address.
259      * There are always 8 pixels to update.
260      * @param addr - address of screen byte. 0 is pixels 0-7 in top left corner
261      * @param pixelByte - on/off for 8 pixels.
262      * @param colorByte - color index for foreground and background.
263      */
264     public void updatePixels(int addr, int pixelByte, int colorByte) {
265         if (addr >= 8000) return;
266         int offset = addr * 8;
267         int bgInx = colorByte & 0x0F;
268         int fgInx = (colorByte & 0xF0) >> 4;
269         int fgColor = palette[fgInx];
270         int bgColor = palette[bgInx];
271         for (int i = 0; i < 8; i++) {
272             if ((pixelByte & 0x80) != 0) {
273                 pixels[offset + i] = fgColor;
274             } else {
275                 pixels[offset + i] = bgColor;
276             }
277             pixelByte <<= 1;
278         }
279         repaint();
280     }
