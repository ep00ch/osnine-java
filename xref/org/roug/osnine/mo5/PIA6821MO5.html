10  /**
11   * 6821 PIA in a Thomsom MO5.
12   * This chip is located at addresses 0xA7C0-0xA7C3.
13   * The MO5 buzzer is controlled by bit 0 on $A7C1. By changing this bit
14   * a quadratic signal is emitted. This is used by the PLAY instruction in BASIC.
15   *
16   * data port A (A7C0)
17   *  bit 0: Switch the screen RAM mapping between pixel and attribute RAMs
18   *  bits 1-4: border color (R,G,B,P)
19   *  bit 5: light pen button
20   *  bit 6: tape drive data output
21   *  bit 7: tape drive data input
22   *  bit 7 is low when no tape drive is plugged, and high when there is one.
23   *  The monitor loading and saving code checks for this to detect the tape drive
24   *
25   * data port B (A7C1)
26   *  bit 0: sound output
27   *  bits 1-3: keyboard column to scan
28   *  bits 4-6: keyboard line to scan
29   *  bit 7: state of key selected by the column and line
30   *
31   * Control ports (A7C2, A7C3)
32   *  CA1: lightpen interrupt (IRQA is wired to 6809 FIRQ)
33   *  CA2: tape drive motor control (output)
34   *  CB1: 50Hz interrupt (IRQB is wired to 6809 IRQ)
35   *  CB2: video incrustation enable (output)
36   */
37  public class PIA6821MO5 extends PIA6821 {

42      /** If PIA has raised IRQ on the bus. */
43      private boolean[] activeIRQ = new boolean[2];

45      /** The event-driven GUI. */
46      private Screen screen;

50      /**
51       * Create PIA and configure the connections to devices.
52       *
53       * @param bus the memory bus to send IRQ and FIRQ signals to.
54       * @param screen the interface to the keyboard
55       * @param tape the interface to tape station
56       */
57      public PIA6821MO5(Bus8Motorola bus, Screen screen, TapeRecorder tape) {
58          super(0xA7C0, bus);
59          this.screen = screen;
60          this.tape = tape;
61          setLayout(1);
62  
63          setIRQCallback(A, (boolean state) -> signalFIRQ(state));
64          setIRQCallback(B, (boolean state) -> signalIRQ(state));
65  
66          setOutputCallback(A, (int mask, int value, int oldValue)
67                     -> screenMemoryBank(mask, value, oldValue));
68          setOutputCallback(B, (int mask, int value, int oldValue)
69                     -> keyboardMatrix(mask, value, oldValue));
70          setControlCallback(A, (boolean state) -> cassetteMotor(state));
71      }

73  
74      private static boolean isBitOn(int x, int n) {
75          return (x & n) != 0;
76      }
77  
78      /**
79       * IRQB is connected to the CPU's IRQ pin.
80       *
81       * @param state - new state of the interrupt.
82       */
83      private void signalIRQ(boolean state) {
84          if (state != activeIRQ[B]) {
85              activeIRQ[B] = state;
86              bus.signalIRQ(state);
87          }
88      }
89  
90      /**
91       * IRQA is connected to the CPU's FIRQ pin.
92       *
93       * @param state - new state of the interrupt.
94       */
95      private void signalFIRQ(boolean state) {
96          if (state != activeIRQ[A]) {
97              activeIRQ[A] = state;
98              bus.signalFIRQ(state);
99          }
100     }

102     /**
103      * Start/stop cassette motor.
104      * If state is false, then turn motor on, otherwise turn off motor.
105      */
106     private void cassetteMotor(boolean state) {
107         tape.cassetteMotor(!state);
108     }

110     /**
111      * Set pixel bank and output to casette.
112      * If C2 is off then the tape motor is running.
113      * The program then sends 0 and 1 at timed intervals to
114      * send values to the tape at 1200 baud.
115      *
116      * @param mask - the data direction mask - 1 = output.
117      * @param value - the new value set in the output registers
118      * @param oldValue - the previous value - for detecting changes.
119      */
120     private void screenMemoryBank(int mask, int value, int oldValue) {
121         screen.setPixelBankActive(isBitOn(value, BIT0));
122 
123         // Write cassette output
124         if (!isC2On(A) && isBitOn(mask, BIT6)) {
125             tape.writeToCassette(isBitOn(value, BIT6));
126         }
127     }

129     /**
130      * Scan keyboard matrix for pressed keys. Set input line B7 if so.
131      * @param mask - the data direction mask - 1 = output.
132      * @param value - the new value set in the output registers
133      * @param oldValue - the previous value - for detecting changes.
134      */
135     private void keyboardMatrix(int mask, int value, int oldValue) {
136         setInputLine(B, 7, !screen.hasKeyPress(value & 0x7E));
137     }
