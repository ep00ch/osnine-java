168     /**
169      * Execute one instruction.
170      */
171     public void execute() {
172         if (resetSignal) {
173             reset();
174         }
175         if (traceInstructions) {
176             disAsm.disasmPC();
177         }
178         ir = fetch();
179 
180         // Select addressing mode
181         switch (ir & 0xf0) {
182             case 0x00: case 0x90: case 0xd0:
183                 mode = DIRECT; break;
184             case 0x20:
185                 mode = RELATIVE; break;
186             case 0x30: case 0x40: case 0x50:
187                 if (ir < 0x34) {
188                     mode = INDEXED;
189                 } else if (ir < 0x38) {
190                     mode = IMMEDIATE;
191                 } else {
192                     mode = INHERENT;
193                 }
194                 break;
195             case 0x60: case 0xa0: case 0xe0:
196                 mode = INDEXED; break;
197             case 0x70: case 0xb0: case 0xf0:
198                 mode = EXTENDED; break;
199             case 0x80: case 0xc0:
200                 mode = IMMEDIATE; break;
201             case 0x10:
202                 switch (ir & 0x0f) {
203                     case 0x02: case 0x03: case 0x09:
204                     case 0x0d: case 0x0e: case 0x0f:
205                         mode = INHERENT; break;
206                     case 0x06: case 0x07:
207                         mode = RELATIVE; break;
208                     case 0x0a: case 0x0c:
209                         mode = IMMEDIATE; break;
210                     case 0x00: case 0x01:
211                         ir <<= 8;
212                         ir |= fetch();
213                         switch (ir & 0xf0) {
214                             case 0x20:
215                                 mode = RELATIVE; break;
216                             case 0x30:
217                                 mode = INHERENT; break;
218                             case 0x80: case 0xc0:
219                                 mode = IMMEDIATE; break;
220                             case 0x90: case 0xd0:
221                                 mode = DIRECT; break;
222                             case 0xa0: case 0xe0:
223                                 mode = INDEXED; break;
224                             case 0xb0: case 0xf0:
225                                 mode = EXTENDED; break;
226                         }
227                         break;
228                 }
229                 break;
230         }
231 
232         // Select instruction
233         if (ir < 0x80) {
234             executeUnder80();
235         } else if (ir >= 0x1080) {
236             execute1080();
237         } else if (ir >= 0x1000) {
238             execute1000();
239         } else {
240             execute80toFF();
241         }
242 
243         if (isNMIActive()) {
244             nmi();
245         }
246         if (isFIRQActive()) {
247             firq();
248         }
249         if (isIRQActive()) {
250             irq();
251         }
252     }

651     private int fetch_operand() {
652         int ret = 0;
653         int addr;
654 
655         switch (mode) {
656             case IMMEDIATE:  // or RELATIVE
657                 ret = fetch();
658                 break;
659             case EXTENDED:
660                 addr = fetch_word();
661                 ret = read(addr);
662                 break;
663             case DIRECT:
664                 addr = (dp.intValue() << 8) | fetch();
665                 ret = read(addr);
666                 break;
667             case INDEXED:
668                 int post = fetch();
669                 do_predecrement(post);
670                 addr = do_effective_address(post);
671                 ret = read(addr);
672                 do_postincrement(post);
673                 break;
674             default:
675                 invalid("addressing mode");
676         }
677 
678         return ret;
679     }

722                 addr = fetch_word();
723                 break;
724             case DIRECT:
725                 addr = dp.intValue() << 8 | fetch();
726                 break;
727             case INDEXED:
728                 post = fetch();
729                 do_predecrement(post);
730                 addr = do_effective_address(post);
731                 do_postincrement(post);
732                 break;
733             default:

739     /**
740      * Calculate indirect addressing.
741      */
742     private int do_effective_address(int post) {
743         int addr = 0;
744         int sOffset;
745         int uOffset;
746 
747         if ((post & 0x80) == 0x00) {
748             addr = refreg(post).intValue() + extend5(post & 0x1f);   // Constant 5-bit offset from register
749         } else {
750             switch (post & 0x1f) {
751                 case 0x00:  // Increment by 1 (done elsewhere)
752                 case 0x02:  // Decrement by 1 (done elsewhere)
753                     addr = refreg(post).intValue();
754                     break;
755                 case 0x01:  // Increment by 2 (done elsewhere)
756                 case 0x03:  // Decrement by 2 (done elsewhere)
757                 case 0x11:  // Increment by 2 (done elsewhere)
758                 case 0x13:  // Decrement by 2 (done elsewhere)
759                     addr = refreg(post).intValue();
760                     break;
761                 case 0x04:  // Non-Indirect No offset
762                 case 0x14:  // Indirect No offset
763                     addr = refreg(post).intValue();
764                     break;
765                 case 0x05:  // Non-Indirect B-register offset
766                 case 0x15:  // Indirect B-register offset
767                     addr = b.getSigned() + refreg(post).intValue();
768                     break;
769                 case 0x06:  // Non-Indirect A-register offset
770                 case 0x16:  // Indirect A-register offset
771                     addr = a.getSigned() + refreg(post).intValue();
772                     break;
773                 case 0x08:  // Non-Indirect 8-bit offset
774                 case 0x18:  // Indirect 8-bit offset
775                     addr = refreg(post).intValue() + extend8(fetch());
776                     break;
777                 case 0x09:  // Non-Indirect 16-bit offset
778                 case 0x19:  // Indirect 16-bit offset
779                     sOffset = getSignedWord(fetch_word());
780                     addr = refreg(post).intValue() + sOffset;
781                     break;
782                 case 0x0b:   // Non-Indirect D-register offset
783                 case 0x1b:   //  Indirect D-register offset
784                     sOffset = refreg(post).getSigned();
785                     addr = (d.intValue() + sOffset) & 0xFFFF;
786                     break;
787                 case 0x0c:   // Non-Indirect Constant 8-bit offset from PC
788                 case 0x1c:   // Indirect Constant 8-bit offset from PC
789                     uOffset = extend8(fetch());
790                     addr = pc.intValue() + uOffset;
791                     break;
792                 case 0x0d:   // Non-Indirect Constant 16-bit offset from PC
793                 case 0x1d:   // Indirect Constant 16-bit offset from PC
794                     sOffset = getSignedWord(fetch_word());
795                     addr = pc.intValue() + sOffset;
796                     break;
797                 case 0x1f:   // Extended indirect
798                     addr = fetch_word();
799                     break;
800                 default:
801                     invalid("indirect addressing postbyte");
802                     break;
803             }
804 
805             // Do extra indirection
806             if ((post & 0x10) != 0) {
807                 addr = read_word(addr);
808             }
809         }
810 
811         return addr;
812     }

917     /**
918      * Add two 8-bit numbers.
919      * The rules for turning on the overflow flag in binary/integer math are two:
920 
921      * 1. If the sum of two numbers with the sign bits off yields a result number
922      *    with the sign bit on, the "overflow" flag is turned on.
923 
924      *    0100 + 0100 = 1000 (overflow flag is turned on)
925 
926      * 2. If the sum of two numbers with the sign bits on yields a result number
927      *    with the sign bit off, the "overflow" flag is turned on.
928 
929      *    1000 + 1000 = 0000 (overflow flag is turned on)
930 
931      * Otherwise, the overflow flag is turned off.
932      * 0100 + 0001 = 0101 (overflow flag is turned off)
933      * 0110 + 1001 = 1111 (overflow flag is turned off)
934      * 1000 + 0001 = 1001 (overflow flag is turned off)
935      * 1100 + 1100 = 1000 (overflow flag is turned off)
936 
937      * Note that you only need to look at the sign bits (leftmost) of the three
938      * numbers to decide if the overflow flag is turned on or off.
939 
940      * If you are doing two's complement (signed) arithmetic, overflow flag on
941      * means the answer is wrong - you added two positive numbers and got a
942      * negative, or you added two negative numbers and got a positive.
943      */
944     private void help_add(UByte refB) {
945         int m = fetch_operand();
946 
947         int halfCarry = (refB.intValue() & 0x0f) + (m & 0x0f);
948         cc.setH(btst(halfCarry, 4));         // Half carry
949 
950         int carryIn = (refB.intValue() & 0x7f) + (m & 0x7f);
951         cc.setV(btst(carryIn, 7));      // Bit 7 carry in
952 
953         int t = refB.intValue() + m;
954         cc.setC(btst(t, 8));
955         cc.setV(cc.getV() != cc.getC());
956         refB.set(t);
957 
958         setBitN(refB);
959         setBitZ(refB);
960     }

1323     /**
1324      * Fast hardware interrupt (FIRQ).
1325      * The <em>fast interrupt request</em> is similar to the IRQ, as it is
1326      * maskable by setting the F bit in the condition code register to 1.
1327      * When an FIRQ is received, only the PC and condition code register are
1328      * saved on the hardware stack. The E bit is not set, because the entire
1329      * machine state has not been saved. The PC for the FIRQ handler is fetched
1330      * from locations FFF6:FFF7. Both the F and the I bits are set to 1 to
1331      * prevent any more interrupts.
1332      *
1333      * The fast interrupt request executes much more quickly than the NMI
1334      * or IRQ, because only three bytes are pushed onto the stack. The FIRQ
1335      * takes ten cycles to execute. The NMI and IRQ require nineteen. The fast
1336      * interrupt request is very useful when speed is essential, but the
1337      * registers are not used extensively. If a reqister is used, it must first
1338      * be pushed and then pulled, before execution of the RTI instruction.
1339      * The RTI restores the condition code register and the PC of the
1340      * interrupted program.
1341      */

1344             cc.setE(0);

1346         }
1347         waitState = false;

1349         cc.setF(1);

1351         pc.set(read_word(FIRQ_ADDR));
1352     }
