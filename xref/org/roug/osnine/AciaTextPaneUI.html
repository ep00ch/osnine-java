118     public void run() {
119         LOGGER.debug("Thread started");
120         try {
121             acia.setDCD(true);
122             try {
123                 while (true) {
124                     int val = acia.waitForValueToTransmit();
125                     sendToGUI(val);
126                     if (Thread.interrupted())
127                         throw new InterruptedException();
128                 }
129             } catch (InterruptedException e) {
130                 LOGGER.error("InterruptException");
131             } catch (Exception e) {
132                 LOGGER.error("Broken connection", e);
133             }
134             acia.setDCD(false);
135         } catch (Exception e) {
136             LOGGER.error("GUI exception");
137         }
138         System.exit(2);
139     }

149     /**
150      * Put the character on the canvas.
151      * GO51 escape codes.
152      *
153      * @param val character to display.
154      */
155     void sendToGUI(int val) throws BadLocationException,IOException {
156         int caretPos = textArea.getCaretPosition();
157         String newchar = Character.toString((char) val);
158         //logCharacter(val, newchar);
159 
160         switch (val) {
161         case 7:
162             Toolkit.getDefaultToolkit().beep();
163             break;
164         case 8:     // Backspace
165             if (backspaceDeletes) {
166                 if (caretPos > 0)
167                     doc.remove(caretPos - 1, 1);
168             } else {
169                 textArea.setCaretPosition(caretPos - 1);
170             }
171             break;
172         case 11:    // Cursor home
173             textArea.setCaretPosition(0);
174             break;
175         case 12:    // Form feed - clear screen
176             doc.remove(0, doc.getLength());
177             currSet = normalSet;
178             break;
179         case 0:  // Ignore NULLs
180         case 3:  // Ignore ETX - End of Text
181         case 13: // Ignore carriage returns
182         case 14: // Ignore command to go to alpha mode.
183         case 127:
184             break;
185         default:
186             int l = textArea.getDocument().getLength();
187             if (caretPos == l) {
188                 doc.insertString(doc.getLength(), newchar, currSet);
189             } else {
190                 doc.insertString(caretPos, newchar, currSet);
191                 //textArea.replaceRange(newchar, caretPos, caretPos + 1);
192             }
193             textArea.setCaretPosition(caretPos + 1);
194         }
195 
196         // Scroll
197         if (getLineCount() > rows) {
198             String t = textArea.getText();
199             textArea.setText(t.substring(t.indexOf('\n') + 1));
200             textArea.setCaretPosition(textArea.getText().length());
201         }
202     }

204     private int getLineCount() throws BadLocationException {
205         int numNL = 0;
206         String t = textArea.getText(0, doc.getLength());
207         int l = t.length();
208         for (int i = 0; i < l; i++) {
209             if (t.charAt(i) == '\n') numNL++;
210         }
211         return numNL;
212     }

236     /**
237      * State engine to handle telnet protocol.
238      */
239     private enum Go51State {
240 
241         NORMAL {
242             @Override
243             Go51State sendToGUI(int val, AciaTextPaneUI h)
244                     throws BadLocationException,IOException {
245                 int caretPos = h.textArea.getCaretPosition();
246                 String newchar = Character.toString((char) val);
247                 switch (val) {
248                 case 7:
249                     Toolkit.getDefaultToolkit().beep();
250                     return NORMAL;
251                 case 8:     // Backspace
252                     if (h.backspaceDeletes) {
253                         if (caretPos > 0)
254                             h.doc.remove(caretPos - 1, 1);
255                     } else {
256                         h.textArea.setCaretPosition(caretPos - 1);
257                     }
258                     return NORMAL;
259                 case 11:    // Cursor home
260                     h.textArea.setCaretPosition(0);
261                     return NORMAL;
262                 case 12:    // Form feed - clear screen
263                     h.doc.remove(0, h.doc.getLength());
264                     h.currSet = h.normalSet;
265                     h.showInverted = false;
266                     h.showUnderline = false;
267                     return NORMAL;
268                 case 0:  // Ignore NULLs
269                 case 3:  // Ignore ETX - End of Text
270                 case 13: // Ignore carriage returns
271                 case 14: // Ignore command to go to alpha mode.
272                 case 127:
273                     return NORMAL;
274                 case 27:
275                     return ESCAPE;
276                 default:
277                     int l = h.textArea.getDocument().getLength();
278                     if (caretPos == l) {
279                         h.doc.insertString(h.doc.getLength(), newchar, h.currSet);
280                     } else {
281                         h.doc.insertString(caretPos, newchar, h.currSet);
282                         //textArea.replaceRange(newchar, caretPos, caretPos + 1);
283                     }
284                     h.textArea.setCaretPosition(caretPos + 1);
285                     return NORMAL;
286                 }
287             }
288         },
289 
290         ESCAPE {
291             @Override
292             Go51State sendToGUI(int val, AciaTextPaneUI h)
293                     throws BadLocationException,IOException {
294                 int caretPos = h.textArea.getCaretPosition();
295                 switch (val) {
296                 case 0x42:  // Clear to end of line
297                     h.doc.remove(caretPos, h.doc.getLength() - caretPos);
298                     return NORMAL;
299                 case 0x46:  // Reverse on
300                     if (h.showUnderline)
301                         h.currSet = h.invertedUnderSet;
302                     else
303                         h.currSet = h.invertedSet;
304                     h.showInverted = true;
305                     return NORMAL;
306                 case 0x47:  // Reverse off
307                     if (h.showUnderline)
308                         h.currSet = h.normalUnderSet;
309                     else
310                         h.currSet = h.normalSet;
311                     h.showInverted = false;
312                     return NORMAL;
313                 case 0x48:  // Underline on
314                     if (h.showInverted)
315                         h.currSet = h.invertedUnderSet;
316                     else
317                         h.currSet = h.normalUnderSet;
318                     h.showUnderline = true;
319                     return NORMAL;
320                 case 0x49:  // Underline off
321                     if (h.showInverted)
322                         h.currSet = h.invertedSet;
323                     else
324                         h.currSet = h.normalSet;
325                     h.showUnderline = false;
326                     return NORMAL;
327                 default:
328                     return NORMAL;
329                 }
330             }
331         };

333         abstract Go51State sendToGUI(int val, AciaTextPaneUI h)
334                 throws BadLocationException,IOException;
335
