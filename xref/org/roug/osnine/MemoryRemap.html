3   /**
4    * Remap memory locations. For instance on the Dragon 64, the interrupt
5    * vectors at $FFF2 and up are mapped to ROM at $BFF2.
6    */
7   public class MemoryRemap extends MemorySegment {
8   
9       /** How many locations to remap. */
10      private int memorySize;
11  
12      /** Relative address for mapping. */
13      private int mapOffset;
14  
15      /** Reference to bus to fetch byte. */
16      private Bus8Motorola bus;
17  
18      /**
19       * Constructor.
20       * The size of the RAM segment is the first argument.
21       * Second argument is the location to remap to. (Don't remap with 0 offset.)
22       *
23       * @param start - First address location of the segment.
24       * @param bus - The bus the segment is attached to.
25       * @param args - additional arguments
26       */
27      public MemoryRemap(int start, Bus8Motorola bus, String... args) {
28          super(start);
29          int size = Integer.decode(args[0]).intValue();
30          this.bus = bus;
31          setEndAddress(start + size);
32          this.memorySize = size;
33          int mapLocation = Integer.decode(args[1]).intValue();
34          mapOffset = mapLocation - start;
35      }
36  
37      @Override
38      protected int load(int addr) {
39          return bus.read(addr + mapOffset);
40      }
41  
42      @Override
43      protected void store(int addr, int val) {
44          bus.write(addr + mapOffset, val);
45      }

47  }
