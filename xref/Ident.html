18      private int read_word(int location) {
19          return (module[location] & 0xff) << 8 | (module[location + 1] & 0xff);

22      private int read_triple(int location) {
23          return (module[location] & 0xff) << 16 | (module[location + 1] & 0xff) << 8 | (module[location + 2] & 0xff);

30      private void printString(String label, int location) {
31          // Get length
32          int length = 0;
33          for (int i = location; i < 0x10000; i++) {
34              if (module[i] <= 0) {
35                  length = i - location + 1;
36                  break;
37              }
38          }
39          module[location + length - 1] &= 0x7f;
40          String buf = new String(module, location, length);
41          module[location + length - 1] |= 0x80;
42          System.out.format("%-20s: %s\n", label, buf);
43      }

45      public Ident(String fileName) throws Exception {
46          FileInputStream inputStream = new FileInputStream(fileName);
47  
48          int size = inputStream.read(module);
49          int pointer = 0;
50          while (pointer < size) {
51              printString("Name", pointer + read_word(pointer + 0x04));
52              printHex("Sync bytes", read_word(pointer + 0));
53              int moduleSize = read_word(pointer + 0x02);
54              printHex("Module size", moduleSize);
55              printHex("Type/lang", read_byte(pointer + 0x06));
56              printHex("Attr/rev", read_byte(pointer + 0x07));
57              printHex("Hdr parity", read_byte(pointer + 0x08));
58              printHex("CRC", read_triple(pointer + moduleSize - 3));
59              int moduleType = read_byte(pointer + 0x06);
60              switch (moduleType) {
61              case 0x11: // Executable module
62              case 0xC1: // Executable system module
63                  printHex("Execution offset", read_word(pointer + 0x09));
64                  printHex("Storage size", read_word(pointer + 0x0B));
65                  break;
66              case 0xC0: // Configuration module
67                  printHex("Top of free RAM", read_triple(pointer + 0x09));
68                  printHex("# IRQ entries", read_byte(pointer + 0x0C));
69                  printHex("# Device entries", read_byte(pointer + 0x0D));
70                  printString("Startup module", pointer + read_word(pointer + 0x0E));
71                  printString("Default storage", pointer + read_word(pointer + 0x10));
72                  printString("Initial path", pointer + read_word(pointer + 0x12));
73                  printString("Bootstrap module", pointer + read_word(pointer + 0x14));
74                  break;
75              case 0xF1: // Device descriptor module
76                  printString("File manager", pointer + read_word(pointer + 0x09));
77                  printString("Device driver", pointer + read_word(pointer + 0x0B));
78                  printHex("Mode byte", read_byte(pointer + 0x0D));
79                  printHex("Device controller addr", read_triple(pointer + 0x0E));
80                  printHex("Init table size", read_byte(pointer + 0x11));
81                  break;
82              }
83              System.out.println();
84              pointer += moduleSize;
85          }
86      }
